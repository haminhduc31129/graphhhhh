\end{tcolorbox}

\end{frame}

\begin{frame}{Bài toán 1 - Đếm thành phần liên thông}

\begin{block}{Đề bài}
Cho một đồ thị vô hướng gồm $n$ đỉnh và $m$ cạnh.  
Hãy đếm số \textbf{thành phần liên thông}.
\end{block}

\pause

\begin{block}{Giới hạn}
\begin{itemize}
    \item $1 \le n \le 10^5$
    \item $1 \le m \le 2 \times 10^5$
\end{itemize}
\end{block}

\pause

\begin{alertblock}{Gợi ý}
Sử dụng DFS hoặc BFS để duyệt từng thành phần.
\end{alertblock}

\end{frame}

% TEST CASE
\begin{frame}[fragile]{Ví dụ mẫu - Bài 1}

\begin{tcolorbox}[title=Input]
\begin{verbatim}
5 3
1 2
2 3
4 5
\end{verbatim}
\end{tcolorbox}

\pause

\begin{tcolorbox}[title=Output]
\begin{verbatim}
2
\end{verbatim}
\end{tcolorbox}

\end{frame}

% SOLUTION 1
\begin{frame}[fragile]{Solution - Bài 1}

\begin{block}{Ý tưởng}
\begin{itemize}
    \item Duyệt qua từng đỉnh
    \item Nếu đỉnh chưa thăm → chạy DFS/BFS
    \item Mỗi lần chạy → tăng số thành phần lên 1
\end{itemize}
\end{block}

\pause

\begin{tcolorbox}[colback=black!5!white,colframe=blue!75!black,title=Mã giả DFS]
\begin{verbatim}
for u in V:
    if not visited[u]:
        DFS(u)
        count += 1
\end{verbatim}
\end{tcolorbox}

\end{frame}

% SOLUTION 3
\begin{frame}[fragile]{Solution - Bài 3}

\begin{block}{Ý tưởng}
\begin{itemize}
    \item Khi DFS từ $u$ sang $v$
    \item Nếu $v$ đã thăm và $v \neq parent[u]$ → có chu trình
\end{itemize}
\end{block}

\pause

\begin{tcolorbox}[colback=black!5!white,colframe=red!70!black,title=Mã giả DFS]
\begin{verbatim}
DFS(u, parent):
    visited[u] = true
    for v in adj[u]:
        if not visited[v]:
            if DFS(v, u): return true
        else if v != parent:
            return true
    return false
\end{verbatim}
\end{tcolorbox}

\end{frame}
