\section{Thuật toán DFS}

\begin{frame}{DFS - Duyệt theo chiều sâu}

    \begin{block}{\textbf{MỞ ĐẦU}}
        Bài viết này sẽ giúp bạn tìm hiểu về cây \textbf{DFS} (Depth First Search Tree – DFS Tree). Cây \textbf{DFS} là một ứng dụng quan trọng có được từ kỹ thuật duyệt đồ thị ưu tiên chiều sâu, giúp giải các bài toán tìm khớp cầu và thành phần liên thông mạnh.
    \end{block}
    \pause
    \textbf{Ứng dụng chuyên sâu:}
    \begin{itemize}
        \item Phát hiện \textbf{chu trình} (Cycle detection)
        \item Phân tích các thành phần\textbf{ liên thông} (Connected)
        \item Sắp xếp \textbf{Topo} (Topological Sort).
        \item Giải nhiều bài toán\textbf{ DP trên cây }kinh điển
        \item Cũng như nhiều ứng dụng khác nữa ...
    \end{itemize}
\end{frame}

\begin{frame}

    \begin{block}{Ý tưởng và cấu trúc}
        \begin{itemize}
            \item Sử dụng \textbf{ngăn xếp (Stack)} hoặc \textbf{Đệ quy}
            \item Thăm một đỉnh kề, sau đó tiếp tục thăm ngay đỉnh kề của đỉnh đó cho đến khi gặp ngõ cụt mới \textbf{quai lui (backtrack).}
        \end{itemize}
    \end{block}
    \pause
    \begin{exampleblock}{Cách hoạt động}
        \begin{itemize}
            \item Bắt đầu từ một đỉnh
            \item Đi sang một đỉnh kề \textbf{chưa thăm}
            \item Lặp lại cho đến khi \textbf{không đi tiếp được}
            \item Quay lui lại đỉnh trước
            \item Tiếp tục với nhánh khác cho đến khi duyệt hết toàn bộ đỉnh của đồ thị
        \end{itemize}
    \end{exampleblock}
\end{frame}

\section{Thuật toán BFS}



\begin{frame}{BFS - Duyệt theo chiều rộng}

    \begin{block}{\textbf{MỞ ĐẦU}}
        Bài viết này sẽ giúp bạn tìm hiểu về cây \textbf{BFS} (Breadth First Search Tree – BFS Tree). Cây \textbf{BFS} là một ứng dụng quan trọng có được từ kỹ thuật duyệt đồ thị theo chiều rộng, giúp giải các bài toán tìm đường đi ngắn nhất trên đồ thị không trọng số và phân tích thành phần liên thông.
    \end{block}
    \pause
    \textbf{Ứng dụng chuyên sâu:}
    \begin{itemize}
        \item Tìm \textbf{đường đi ngắn nhất} (Shortest Path - unweighted)
        \item Phân tích các thành phần \textbf{liên thông} (Connected)
        \item Kiểm tra đồ thị \textbf{hai phía} (Bipartite)
        \item Lan truyền theo \textbf{từng lớp} (Level Order Traversal)
        \item Cũng như nhiều ứng dụng khác nữa ...
    \end{itemize}
\end{frame}

\begin{frame}

    \begin{block}{Ý tưởng và cấu trúc}
        \begin{itemize}
            \item Sử dụng \textbf{hàng đợi (Queue)}
            \item Thăm tất cả các đỉnh kề của một đỉnh trước khi đi tiếp sang các đỉnh ở mức sâu hơn
        \end{itemize}
    \end{block}
    \pause
    \begin{exampleblock}{Cách hoạt động}
        \begin{itemize}
            \item Bắt đầu từ một đỉnh
            \item Đưa đỉnh đó vào \textbf{queue}
            \item Lấy đỉnh ra và thăm tất cả các đỉnh kề \textbf{chưa thăm}
            \item Đưa các đỉnh đó vào queue
            \item Lặp lại cho đến khi queue rỗng
        \end{itemize}
    \end{exampleblock}
\end{frame}

        \begin{itemize}

        \end{itemize}
